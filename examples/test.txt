DEVICES [
    G1 = NAND 2;
    SW1, SW2 = SWITCH 0;
    G1 = AND 2;
    D1 = DTYPE;
    SW1, SW2, SW3 = SWITCH 0;
]

CONNECTIONS [
    device G1 {
        SW1 -> G1.I1;
        G2 -> G1.I2;
    }
    device G2 {
        SW2 -> G2.I1;
        G1 -> G2.I2;
     }
    device D1 {
        SW1 -> D1.DATA;
        SW2 -> D1.CLK;
        SW3 -> D1.SET;
        G1 -> D1.CLEAR;
    }
]

MONITORS [
   all;
]
